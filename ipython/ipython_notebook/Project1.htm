def OneMax(x):
    return x.count("1")
 
assert OneMax("1001001") == 3
 
def LeadingOnes(x):
    if x and x[0] == "1": 
        return LeadingOnes(x[1:]) + 1
    return 0
 
assert LeadingOnes("110010011") == 2
 
def Jump(k):
    def Jumpk(x):
        n = len(x)
        assert k < n, "Jumpk is not defined"
        _x_ = OneMax(x)
        if _x_ < n - k:
            return _x_
        if n - k <= _x_ < n:
            return n - k
        return n
    Jumpk.__name__ = "Jump_{}".format(k)
    return Jumpk
 
assert Jump(2)("11000") == 2
assert Jump(2)("11110") == 3
assert Jump(2)("11111") == 5
 
def BinVal(x):
    return int("".join(x), 2)
 
assert BinVal("1100") == 12
            
def RoyalRoads(k):
    def RoyalRoadsk(x):
        n = len(x)
        assert n % k == 0, "RoyalRoadsk is not defined"
        return sum(all( char == "1" for char in x[i:i + k]) for i in range(0, n, k))
    RoyalRoadsk.__name__ = "RoyalRoads_{}".format(k)
    return RoyalRoadsk
 
assert RoyalRoads(3)("111000000") == 1
assert RoyalRoads(3)("110111110") == 1

import random
 
def RandomString(n):
    return list(bin(random.getrandbits(n))[2:].rjust(n, "0"))
 
assert len(RandomString(5)) == 5
 
def GreaterOrEqual(v1, v2):
    return v1 >= v2
 
def Greater(v1, v2):
    return v1 > v2
 
def Flip(x, i):
    x[i] = ("1" if x[i] == "0" else "0")
 
def RLS(f, n, stopped, better):
    x = RandomString(n)
    while not stopped(x):
        y = x[:]
        i = random.randint(0, n - 1)
        Flip(y, i)
        if better(f(y), f(x)):
            x = y
        yield x
    return x
 
def EA_1_plus_1(f, n, stopped, better):
    x = RandomString(n)
    while not stopped(x):
        y = x[:]
        for i in range(n):
            if random.random() < 1 / n:
                Flip(y, i)
        if better(f(y), f(x)):
            x = y
        yield x
    return x
